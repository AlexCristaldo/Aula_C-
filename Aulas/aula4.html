Matrizes

Nesta aula serão estudadas as matrizes, estruturas bidimensionais compostas por linhas e colunas.

Vídeo da Aula



  • Declarar matrizes.
  • Preencher linhas e colunas.
  • Percorrer matrizes.
  • Realizar operações com matrizes.

Matrizes são conjuntos organizados em linhas e colunas. São utilizadas em diversas áreas, como computação gráfica, inteligência artificial e processamento de imagens.

Curiosidade:

Jogos digitais utilizam matrizes para representar mapas, tabuleiros e ambientes virtuais.

Exercícios

Exercício 1

Leia uma matriz 2x2 e mostre todos os seus elementos.

Ver Resolução 
    
#include 
using namespace std;

int main() {

    int mat[2][2];

    for(int i = 0; i < 2; i++){
        for(int j = 0; j < 2; j++){
            cin >> mat[i][j];
        }
    }

    for(int i = 0; i < 2; i++){
        for(int j = 0; j < 2; j++){
            cout << mat[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}

Exercício 2

Leia uma matriz 3x3 e calcule a soma de todos os elementos.

Ver Resolução 

#include 
using namespace std;

int main() {

    int mat[3][3];
    int soma = 0;

    for(int i = 0; i < 3; i++){
        for(int j = 0; j < 3; j++){

}

Exercício 4

Leia uma matriz 3x3 e conte quantos números pares existem nela.

Ver Resolução 

#include 
using namespace std;

int main() {

    int mat[3][3];
    int pares = 0;

    for(int i = 0; i < 3; i++){
        for(int j = 0; j < 3; j++){
            cin >> mat[i][j];

            if(mat[i][j] % 2 == 0){
                pares++;
            }
        }
    }

    cout << "Quantidade de pares: " << pares;

    return 0;
}

Exercício 5

Leia uma matriz 3x3 e encontre o maior elemento.

Ver Resolução 

#include 
using namespace std;

int main() {

    int mat[3][3];

    for(int i = 0; i < 3; i++){
        for(int j = 0; j < 3; j++){
            cin >> mat[i][j];
        }
    }

    int maior = mat[0][0];

    for(int i = 0; i < 3; i++){
        for(int j = 0; j < 3; j++){

            if(mat[i][j] > maior){
                maior = mat[i][j];
            }

        }
    }

    cout << "Maior valor: " << maior;

    return 0;
}

Exercício 6

Leia duas matrizes 2x2 e crie uma terceira matriz contendo a soma das duas.

Ver Resolução 

#include 
using namespace std;

int main() {

    int A[2][2];
    int B[2][2];
    int C[2][2];

    for(int i = 0; i < 2; i++){
        for(int j = 0; j < 2; j++){
            cin >> A[i][j];
        }
    }

    for(int i = 0; i < 2; i++){
        for(int j = 0; j < 2; j++){
            cin >> B[i][j];
        }
    }

    for(int i = 0; i < 2; i++){
        for(int j = 0; j < 2; j++){
            C[i][j] = A[i][j] + B[i][j];
        }
    }

    cout << "Matriz soma:" << endl;

    for(int i = 0; i < 2; i++){
        for(int j = 0; j < 2; j++){
            cout << C[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}
