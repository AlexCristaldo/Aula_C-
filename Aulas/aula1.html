Exercício 1

Crie um programa que leia dois números e mostre a soma.

Ver Resolução 
#include <iostream>
using namespace std;

int main() {

    int a, b;

    cin >> a;
    cin >> b;

    cout << a + b;

    return 0;
}

Exercício 2

Crie um programa que leia a idade de uma pessoa e mostre a idade informada.

Ver Resolução 
    
#include <iostream>
using namespace std;

int main() {

    int idade;

    cout << "Digite sua idade: ";
    cin >> idade;

    cout << "Sua idade e: " << idade;

    return 0;
}

Exercício 3

Crie um programa que leia dois números e mostre a soma, subtração, multiplicação e divisão.

Ver Resolução 

#include <iostream>
using namespace std;

int main() {

    float a, b;

    cin >> a;
    cin >> b;

    cout << "Soma: " << a + b << endl;
    cout << "Subtracao: " << a - b << endl;
    cout << "Multiplicacao: " << a * b << endl;
    cout << "Divisao: " << a / b << endl;

    return 0;
}
