Exercício 1

Leia um número e informe se ele é positivo ou negativo.

Ver Resolução 

#include 
using namespace std;

int main() {

    int num;

    cin >> num;

    if(num >= 0){
        cout << "Positivo";
    }
    else{
        cout << "Negativo";
    }

    return 0;
}

Exercício 3

Leia uma nota de 0 a 10 e mostre: Aprovado (nota ≥ 6) Recuperação (nota ≥ 4 e < 6) Reprovado (nota < 4)

Ver Resolução 

#include 
using namespace std;

int main() {

    float nota;

    cin >> nota;

    if(nota >= 6){
        cout << "Aprovado";
    }
    else if(nota >= 4){
        cout << "Recuperacao";
    }
    else{
        cout << "Reprovado";
    }

    return 0;
}

Exercício 4

Mostre os números de 1 até 10 usando um "for".

Ver Resolução 

#include 
using namespace std;

int main() {

    for(int i = 1; i <= 10; i++){
        cout << i << endl;
    }

    return 0;
}

Exercício 5

Leia um número e mostre sua tabuada de 1 a 10.

Ver Resolução 

#include 
using namespace std;

int main() {

    int num;

    cin >> num;

    for(int i = 1; i <= 10; i++){
        cout << num << " x " << i
             << " = " << num * i << endl;
    }

    return 0;
}

Exercício 6

Mostre os números de 10 até 1 utilizando while.

Ver Resolução 

#include 
using namespace std;

int main() {

    int i = 10;

    while(i >= 1){
        cout << i << endl;
        i--;
    }

    return 0;
}
